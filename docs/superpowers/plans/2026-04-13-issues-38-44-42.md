```rust
fn chunk_document(
    document: &EvalDocument,
    chunking: &ChunkingStrategy,
) -> EvalResult<Vec<ChunkRecord>> {
    use fastrag_core::{Document, Element, ElementKind, FileFormat, Metadata};

    // Build a minimal Document from the flat EvalDocument.
    // For ByTitle, create a Title element so by_title_chunk splits on it.
    let elements = match (&document.title, chunking) {
        (Some(title), ChunkingStrategy::ByTitle { .. }) if !title.is_empty() => {
            vec![
                Element::new(ElementKind::Title, title.as_str()),
                Element::new(ElementKind::Paragraph, document.text.as_str()),
            ]
        }
        _ => {
            let mut text = String::new();
            if let Some(title) = &document.title {
                if !title.is_empty() {
                    text.push_str(title);
                    text.push_str("\n\n");
                }
            }
            text.push_str(&document.text);
            vec![Element::new(ElementKind::Paragraph, text)]
        }
    };

    let doc = Document {
        metadata: Metadata::new(FileFormat::Text),
        elements,
    };

    let chunks = doc.chunk(chunking);
    Ok(chunks
        .into_iter()
        .enumerate()
        .map(|(i, c)| ChunkRecord {
            doc_id: document.id.clone(),
            title: document.title.clone(),
            text: c.text,
            chunk_index: i,
        })
        .collect())
}
```

```rust
pub fn matches(expr: &FilterExpr, fields: &[(String, TypedValue)]) -> bool {
    match expr {
        FilterExpr::Eq { field, value } => lookup(fields, field).map_or(false, |v| v == value),
        FilterExpr::Neq { field, value } => lookup(fields, field).map_or(true, |v| v != value),
        FilterExpr::Gt { field, value } => cmp_ord(fields, field, value, |o| o == std::cmp::Ordering::Greater),
        FilterExpr::Gte { field, value } => cmp_ord(fields, field, value, |o| o != std::cmp::Ordering::Less),
        FilterExpr::Lt { field, value } => cmp_ord(fields, field, value, |o| o == std::cmp::Ordering::Less),
        FilterExpr::Lte { field, value } => cmp_ord(fields, field, value, |o| o != std::cmp::Ordering::Greater),
        FilterExpr::In { field, values } => lookup(fields, field).map_or(false, |v| values.contains(v)),
        FilterExpr::NotIn { field, values } => lookup(fields, field).map_or(true, |v| !values.contains(v)),
        FilterExpr::Contains { field, value } => {
            lookup(fields, field).map_or(false, |v| match v {
                TypedValue::Array(arr) => arr.contains(value),
                _ => v == value,
            })
        }
        FilterExpr::All { field, values } => {
            lookup(fields, field).map_or(false, |v| match v {
                TypedValue::Array(arr) => values.iter().all(|val| arr.contains(val)),
                _ => false,
            })
        }
        FilterExpr::And(exprs) => exprs.iter().all(|e| matches(e, fields)),
        FilterExpr::Or(exprs) => exprs.iter().any(|e| matches(e, fields)),
        FilterExpr::Not(inner) => !matches(inner, fields),
    }
}

fn lookup<'a>(fields: &'a [(String, TypedValue)], name: &str) -> Option<&'a TypedValue> {
    fields.iter().find(|(n, _)| n == name).map(|(_, v)| v)
}

fn cmp_ord(
    fields: &[(String, TypedValue)],
    field: &str,
    value: &TypedValue,
    pred: impl Fn(std::cmp::Ordering) -> bool,
) -> bool {
    let Some(actual) = lookup(fields, field) else {
        return false;
    };
    match (actual, value) {
        (TypedValue::Numeric(a), TypedValue::Numeric(b)) => a.partial_cmp(b).map_or(false, &pred),
        (TypedValue::Date(a), TypedValue::Date(b)) => pred(a.cmp(b)),
        (TypedValue::String(a), TypedValue::String(b)) => pred(a.cmp(b)),
        _ => false, // Bool and Array don't support ordering
    }
}
```

```rust
/// Fetch metadata (user fields) for the given internal IDs.
///
/// Returns `(id, Vec<(field_name, TypedValue)>)` for each found document.
pub fn fetch_metadata(
    &self,
    ids: &[u64],
) -> StoreResult<Vec<(u64, Vec<(String, TypedValue)>)>> {
    let docs = self.fetch_by_ids(ids)?;
    let mut results = Vec::with_capacity(docs.len());

    for doc in &docs {
        let id = doc
            .get_first(self.core.id)
            .and_then(|v| v.as_u64())
            .unwrap_or(0);

        let mut fields = Vec::new();
        for handle in &self.user_fields {
            if let Some(value) = extract_typed_value(doc, handle) {
                fields.push((handle.name.clone(), value));
            }
        }
        results.push((id, fields));
    }
    Ok(results)
}
```

```rust
use crate::schema::TypedValue;

fn extract_typed_value(doc: &TantivyDocument, handle: &UserFieldHandle) -> Option<TypedValue> {
    match handle.typed {
        TypedKind::String => doc
            .get_first(handle.field)
            .and_then(|v| v.as_str())
            .map(|s| TypedValue::String(s.to_string())),
        TypedKind::Numeric => doc
            .get_first(handle.field)
            .and_then(|v| v.as_f64())
            .map(TypedValue::Numeric),
        TypedKind::Bool => doc
            .get_first(handle.field)
            .and_then(|v| v.as_u64())
            .map(|n| TypedValue::Bool(n != 0)),
        TypedKind::Date => doc
            .get_first(handle.field)
            .and_then(|v| v.as_date())
            .and_then(|dt| {
                let ts = dt.into_timestamp_secs();
                chrono::DateTime::from_timestamp(ts, 0)
                    .map(|d| TypedValue::Date(d.date_naive()))
            }),
        TypedKind::Array => {
            let values: Vec<TypedValue> = doc
                .get_all(handle.field)
                .filter_map(|v| {
                    v.as_str()
                        .map(|s| TypedValue::String(s.to_string()))
                        .or_else(|| v.as_f64().map(TypedValue::Numeric))
                })
                .collect();
            if values.is_empty() {
                None
            } else {
                Some(TypedValue::Array(values))
            }
        }
    }
}
```
